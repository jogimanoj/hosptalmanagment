foreign key

CREATE TABLE lab_address (
    address_id INT AUTO_INCREMENT PRIMARY KEY,
    address     VARCHAR(50),
    city        VARCHAR(50),
    state       VARCHAR(50),
    landmark    VARCHAR(50),
    pincode     VARCHAR(6)
);


ALTER TABLE Laboratory
ADD COLUMN lab_address INT,
ADD CONSTRAINT fk_address
    FOREIGN KEY (lab_address)
    REFERENCES address(lab_address)
    ON DELETE SET NULL
    ON UPDATE CASCADE;
